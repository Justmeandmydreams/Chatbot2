# bot_adapter.py — connects to Teams, WhatsApp, Web Chat
from botbuilder.core import BotFrameworkAdapter, TurnContext
from botbuilder.schema import Activity
import aiohttp

adapter = BotFrameworkAdapter({
    "appId": "YOUR_BOT_APP_ID",
    "appPassword": "YOUR_BOT_APP_PASSWORD"
})

async def on_message(turn_context: TurnContext):
    user_msg = turn_context.activity.text
    # Forward to your FastAPI /chat endpoint
    async with aiohttp.ClientSession() as session:
        async with session.post("https://college-chatbot-api.azurewebsites.net/chat",
                                json={"message": user_msg}) as r:
            data = await r.json()
    await turn_context.send_activity(data["reply"])
